****************************************** ENTITIES ******************************************

USER
- email: PK
- firstName: string
- lastName: string
- password: string *** (TODO) ***
- type: Enum(MANAGER, EMPLOYEE)

PRODUCT
- id: PK
- name: string
- description: string
- category_id: FK

VARIANT
- id: PK
- name: string
- price: float
- product_id: FK

CATEGORY
- id: PK
- name: string

OPTION
- id: PK
- name: string
- type: Enum(DROPDOWN, MULTISELECT)

OPTIONVALUE
- id: PK
- name: string
- option_id: FK

PRODUCT_OPTION (Join Table)
- id: PK
- product_id: FK
- option_id: FK

ORDER
- id: PK
- status: Enum(UNFULFILLED, FULFILLED)

ORDERLINE
- id: PK
- order_id: FK
- quantity: int

ORDERLINE_OPTIONVALUE (Join Table)
- id: PK
- orderline_id: FK
- optionvalue_id: FK

****************************************** API ******************************************

GET Items
- Args:
    - first: int
    - from: int
    - category: item_id!
- Response:
    - Array<Item>

GET Orders
- Args:
    - first: int
    - from: int
- Response:
    - Array<Order>

GET Order
- Args:
    - id: order_id!
- Response: 
    - Order

POST CreateOrder
- Args:
    - items: Array<OrderItems>
- Response:
    - Order

POST UpdateOrder
- Args:
    - id: ID!
    - status: Enum(UNFULFILLED, FULFILLED)
- Response:
    - Order

Type OrderItems = {
    item_id: ID
    quantity: int
    options: Array<OptionItems>
}

Type OptionItems = {
    option_id: ID
    values: Array<string>
}

GET Categories
- Args:
    - first: int
    - from: int
- Response:
    - Array<Category>

POST Category
- Args:
    - id: ID!
    - name: string
- Response:
    - Category

DELETE Category
- Args:
    - id: ID!
- Response:
    - isDeleted: bool

GET Users
- Args:
    - first: int
    - from: int
- Response:
    - Array<User>

POST User
- Args:
    - email: string
    - firstName: string
    - lastName: string
    - password: string
    - type: Enum(MANAGER, EMPLOYEE)
- Response:
    - User

DELETE User
- Args:
    - id: ID!
- Response:
    - isDeleted: bool
